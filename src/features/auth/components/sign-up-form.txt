'use client';

import { useRouter } from 'next/navigation';
import { useActionState, useEffect } from 'react';
import { signUp } from '../actions/sign-up.action';
import type { AuthState } from '../types';
import SubmitButton from './submit-button';

export default function SignUpForm() {
    const router = useRouter()
    const [state, formAction] = useActionState<AuthState, FormData>(signUp, null)

    useEffect(() => {
        if (state?.isAuthenticated && state?.redirect) {
            router.push(state.redirect)
        }
    }, [state, router])

    return (
        <form action={formAction} className="space-y-4">
            <div>
                <label 
                    htmlFor="email" 
                    className="block text-sm font-medium text-zinc-200"
                >
                    Email
                </label>
                <input
                    id="email"
                    name="email"
                    type="email"
                    required
                    className="mt-1 block w-full rounded-md border border-zinc-600 bg-zinc-800 px-3 py-2 text-white shadow-sm focus:border-white focus:outline-none focus:ring-1 focus:ring-white sm:text-sm"
                />
                {state?.error?.email && (
                    <p className="mt-1 text-sm text-red-500">{state.error.email[0]}</p>
                )}
            </div>

            <div>
                <label 
                    htmlFor="password" 
                    className="block text-sm font-medium text-zinc-200"
                >
                    Password
                </label>
                <input
                    id="password"
                    name="password"
                    type="password"
                    required
                    className="mt-1 block w-full rounded-md border border-zinc-600 bg-zinc-800 px-3 py-2 text-white shadow-sm focus:border-white focus:outline-none focus:ring-1 focus:ring-white sm:text-sm"
                />
                {state?.error?.password && (
                    <p className="mt-1 text-sm text-red-500">{state.error.password[0]}</p>
                )}
            </div>

            <div>
                <label 
                    htmlFor="confirmPassword" 
                    className="block text-sm font-medium text-zinc-200"
                >
                    Confirm Password
                </label>
                <input
                    id="confirmPassword"
                    name="confirmPassword"
                    type="password"
                    required
                    className="mt-1 block w-full rounded-md border border-zinc-600 bg-zinc-800 px-3 py-2 text-white shadow-sm focus:border-white focus:outline-none focus:ring-1 focus:ring-white sm:text-sm"
                />
                {state?.error?.confirmPassword && (
                    <p className="mt-1 text-sm text-red-500">
                        {state.error.confirmPassword[0]}
                    </p>
                )}
            </div>

            <SubmitButton variant="signup" />
        </form>
    )
}
